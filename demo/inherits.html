QST.inherits

    function human(name) {
        this.name = name;
    }
    human.prototype.eat = function(){
        $('.con').html($('.con').html()+"<br/>"+this.name+"开始吃饭");
    };

    function chinese(name) {
        this.name = name+"(中国人)";
    }
    QST.inherits(chinese,human);

    var fishow = new human('fishow')
    var ares = new chinese('ares');

    $('.click1').bind('click',function(){
        fishow.eat();
    })
    $('.click2').bind('click',function(){
        ares.eat();
    })
